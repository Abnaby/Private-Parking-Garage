223void LCD_voidGotoXY(LCD_Config* ptrCnfg , u8 x , u8 y)
224{
225 /*
226 -----------
227 | DDRAM | 0x00
228 | | for Row 0 and in between 0x00 -> 0x0f select Col
229 | |
230 | | 0x0f
231
232 | DDRAM | 0x40
233 | | for Row 1 and in between 0x40 -> 0x4f select Col
234 | |
235 | | 0x4f
236 */
237 u8 address = 0 ;
238 if( y < ACCESS_STRUCT(ptrCnfg,LCD_SIZE_Rows) && x < ACCESS_STRUCT(ptrCnfg,LCD_SIZE_Cols) )
239 {
240 /* Using Set DDRAM Address cmd
241 * RS 0
242 * RW 0 --> 0 by default
243 * D7 1
244 * ADD
245 */
246 address = y * 0x40 + x; /* If 0 --> Row 1 , If 0x40 Row 2 */
247 address = address ; /* To Goto x position */
248 address |= (1 << 7) ;
249
250 LCD_voidWriteCmd(ptrCnfg , address);
251
252
253 }

262void LCD_voidStoreCustomChar(LCD_Config* ptrLCD ,u8* ptrToArray, u8 copy_u8CharIndex)
263{
264 u8 address = 0 ,index ;
265 /* Have 8 Places Only for Custom Char Can Store in CGRAM */
266 if(copy_u8CharIndex < 8 )
267 {
268 address = copy_u8CharIndex * 0x08 ;
269 // To Make DDRAM point to CGROM
270 /*
271 * Set CGRAM address
272 * RS = 0
273 * RW = 0
274 * D7 = 0
275 * D6 = 1
276 * D0:D6 Address
277 */
278 address |= (1<<6) ;
279 LCD_voidWriteCmd(ptrLCD,address);
280 for(index = 0 ; index < 8 ; index++)
281 {
282 LCD_voidSendChar(ptrLCD,ptrToArray[index]);
283 }
284 // Restore point to DDRAM OPTIONAL by default display from DDRAM
285 LCD_voidWriteCmd(ptrLCD,LCD_RETURN_HOME);
286
287 }
