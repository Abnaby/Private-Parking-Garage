274void ECU1_Entance_APP_LOOP(void)
275{
276 // Stuck Here Until Get Valid ID
277 while(Glob_ID_Valid != VALID_ID)
278 {
279 GateControl_voidCloseGate() ;
280 }
281 // If Glob_ID_Valid == VALID_ID

291 /* Renable Interrupts */
