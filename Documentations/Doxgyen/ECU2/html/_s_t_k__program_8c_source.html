292void SysTick_Handler(void) {
293 u8 Local_u8Temporary = 0;
294 if (STK_u8ModeOfInterval == STK_SINGLE_INTERVAL)
295 {
296 /* Disable STK Interrupt */
297 CLR_BIT(STK->CTRL, 1);
298 /* Stop Timer */
299 CLR_BIT(STK->CTRL, 0);
300 STK -> LOAD = 0;
301 STK -> VAL = 0;
302 }
303 /* Callback notification */
304 Glob_STKCallback();
305 /* Clear interrupt flag */
306 Local_u8Temporary = GET_BIT(STK->CTRL,16);
307}
