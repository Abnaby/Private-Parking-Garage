235void LCD_voidGotoXY(LCD_Config* ptrCnfg , u8 x , u8 y)
236{
237 /*
238 -----------
239 | DDRAM | 0x00
240 | | for Row 0 and in between 0x00 -> 0x0f select Col
241 | |
242 | | 0x0f
243
244 | DDRAM | 0x40
245 | | for Row 1 and in between 0x40 -> 0x4f select Col
246 | |
247 | | 0x4f
248 */
249 u8 address = 0 ;
250 if( y < ACCESS_STRUCT(ptrCnfg,LCD_SIZE_Rows) && x < ACCESS_STRUCT(ptrCnfg,LCD_SIZE_Cols) )
251 {
252 /* Using Set DDRAM Address cmd
253 * RS 0
254 * RW 0 --> 0 by default
255 * D7 1
256 * ADD
257 */
258 if(y < 2 )
259 {
260 address = y * 0x40 + x; /* If 0 --> Row 1 , If 0x40 Row 2 */
261 }
262 else
263 {
264 address = (y-2) * 0x40 + 0x10 + x;
265
266 }
267 address |= (1 << 7) ;
268
269
270 LCD_voidWriteCmd(ptrCnfg , address);
271
272
273 }

282void LCD_voidStoreCustomChar(LCD_Config* ptrLCD ,u8* ptrToArray, u8 copy_u8CharIndex)
283{
284 u8 address = 0 ,index ;
285 /* Have 8 Places Only for Custom Char Can Store in CGRAM */
286 if(copy_u8CharIndex < 8 )
287 {
288 address = copy_u8CharIndex * 0x08 ;
289 // To Make DDRAM point to CGROM
290 /*
291 * Set CGRAM address
292 * RS = 0
293 * RW = 0
294 * D7 = 0
295 * D6 = 1
296 * D0:D6 Address
297 */
298 address |= (1<<6) ;
299 LCD_voidWriteCmd(ptrLCD,address);
300 for(index = 0 ; index < 8 ; index++)
301 {
302 LCD_voidSendChar(ptrLCD,ptrToArray[index]);
303 }
304 // Restore point to DDRAM OPTIONAL by default display from DDRAM
305 LCD_voidWriteCmd(ptrLCD,LCD_RETURN_HOME);
306
307 }
