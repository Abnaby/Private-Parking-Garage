68static void I2C_voidTimingInit(I2C_Selection_t I2Cx, u32 PClock , I2C_ClockSpeed_t I2C_Clock_Speed)
69{
70 /* Get Whole ICR2 Reg Value */
71 u16 I2C_Temp_Reg = ACCESS_I2Cx(I2Cx)->CR2 ;
72 I2C_Temp_Reg &= ~ (I2C_CR2_FREQ_Msk) ;
73 /* SET FREQ BASED ON PCLOCK1 FREQ */
74 u16 FreqRange = PClock / 1000000 ;
75 I2C_Temp_Reg |= FreqRange ;
76 ACCESS_I2Cx(I2Cx)->CR2 = I2C_Temp_Reg ;
77 /* Config I2C_CCR */
78 ACCESS_I2Cx(I2Cx)->CR1 &=~(I2C_CR1_PE);
79 /* Config Standert Mode */
80 I2C_Temp_Reg = 0 ;
81 if(I2C_Clock_Speed == I2C_STANDART_MODE_50K_HZ || I2C_Clock_Speed == I2C_STANDART_MODE_100K_HZ)
82 {
83 // CCR CALC
84 // Tclk/2 = CRR * Tpclk1
85 // CCR = Tclk / (2 * Tpclk1)
86 // CRR = Fpclk1 / (2 * I2C_Clock_freq )
87
88 u16 result = (u16)(PClock /(I2C_Clock_Speed <<1 )); // ( << 1 ) = * 2
89 I2C_Temp_Reg |= result;
90 ACCESS_I2Cx(I2Cx)->CCR = I2C_Temp_Reg ;
91 /*-------------------------- Configure Rise time register --------------*/
92 I2C_Temp_Reg = 0 ;
93 result = (u16)((PClock/1000000) )+1;
94 I2C_Temp_Reg |= result;
95 ACCESS_I2Cx(I2Cx)->TRISE = I2C_Temp_Reg ;
96
97 }
98 else
99 {
100 // <TODO ERROR> FAST MODE NOT SUPORTTED
101 }
102}

112{

115 /* Add Global Var */
116 if(I2Cx == (u32)I2C1)

136 I2C_TEMP_REG |= (I2C_UserStruct->I2C_ClockStretch)

155 ACCESS_I2Cx(I2Cx)->OAR1 |= I2C_TEMP_REG ;

192void I2C_voidMasterTransmitDMA(I2C_Selection_t I2Cx , u16 copy_u16SlaveAddress , I2C_startState_t I2C_startState)
193{
194
195 volatile u32 LOC_u32DummyVar = 0 ;
196 LOC_u32DummyVar++; // to ignore warning
197
198 // 1. Set the START bit in I2C_CR1 register to generate start Condition
199 I2C_voidGenerateStart(I2Cx, I2C_startState);
200
201 // 2. Wait for EV5 5: SB=1, cleared by reading SR1 register followed by writing DR register with Address.
202 while(!(I2C_GetFlags(I2Cx , I2C_EV5))) ;
203 LOC_u32DummyVar = ACCESS_I2Cx(I2Cx)->SR1 ;
204
205 // 3. Send Address
206 I2C_voidSendAddress( I2Cx , copy_u16SlaveAddress , I2C_MASTER_TX);
207
208 // 4. Wait for EV6: ADDR=1, cleared by reading SR1 register followed by reading SR2.
209 while( !(I2C_GetFlags( I2Cx, I2C_EV6)));
210
211 // 5. Check on [ Busy / MSL / TXE / TRA ] flags from SR1 and SR2 "Clear ADDR Flag"
212 while(!(I2C_GetFlags( I2Cx, MASTER_BYTE_TRANSMITING)));

218

220 u16 LOC_u16Counter = 0 ;
221 volatile u32 LOC_u32DummyVar = 0 ;
222 LOC_u32DummyVar++; // to ignore warning
223
224 // 1. Set the START bit in I2C_CR1 register to generate start Condition
225 I2C_voidGenerateStart(I2Cx, I2C_startState);
226
227 // 2. Wait for EV5 5: SB=1, cleared by reading SR1 register followed by writing DR register with Address.
228 while(!(I2C_GetFlags(I2Cx , I2C_EV5))) ;
229 LOC_u32DummyVar = ACCESS_I2Cx(I2Cx)->SR1 ;

238 while(!(I2C_GetFlags( I2Cx, MASTER_BYTE_TRANSMITING)));
239
240 // 6- Send Data
241 for(LOC_u16Counter = 0 ; LOC_u16Counter < copy_u16DataLength ; LOC_u16Counter++)
242 {
243 ACCESS_I2Cx(I2Cx)->DR = ptr_u8Data[LOC_u16Counter] ;
244 // 7- Wait EV8: TxE=1, shift register not empty, data register empty, cleared by writing DR register
245 while(!(I2C_GetFlags( I2Cx, I2C_EV8)));
246 }
247 // 8. Wait EV8_2: TxE=1, BTF = 1, Program Stop request. TxE and BTF are cleared by hardware by the Stop condition
248 while(!(I2C_GetFlags( I2Cx, I2C_EV8_2)));
249 while(!(I2C_GetFlags( I2Cx, I2C_EV8_1)));

253 I2C_voidGenerateStop(I2Cx);

261 LOC_u32DummyVar++; // to ignore warning
262
263 // 1. Set the START bit in I2C_CR1 register to generate start Condition
264 I2C_voidGenerateStart(I2Cx, I2C_startState);
265
266 // 2. Wait for EV5 5: SB=1, cleared by reading SR1 register followed by writing DR register with Address.

272
273 // 4. Wait for EV6: ADDR=1, cleared by reading SR1 register followed by reading SR2.
274 while( !(I2C_GetFlags( I2Cx, I2C_EV6)));
275
276 // 5. ACK
277 I2C_RETURN_WITH_ACK();
278
279 // 6. Receive Data
280 for(LOC_u16Counter = 0 ; LOC_u16Counter < copy_u16DataLength ; LOC_u16Counter++)
281 {
282 // Wait EV7: RxNE=1 Cleared by reading DR register
283 while( !(I2C_GetFlags( I2Cx, I2C_EV7)));
284 ptr_u8Data[LOC_u16Counter] = ACCESS_I2Cx(I2Cx)->DR ;
285 }
286 // 7- Send NACK after received all data
287 I2C_RETURN_NACK() ;
288
289 // 8- Chk Stop
290 if(I2C_stopState == I2C_END_WITH_STOP)
291 I2C_voidGenerateStop(I2Cx);

316I2C_FlagsState_t I2C_GetFlags(I2C_Selection_t I2Cx , I2C_Flags_t I2C_Flag)
317{
318 I2C_FlagsState_t LOC_bitstatus = I2C_FLAG_RESET;
319 u32 LOC_u32Reg = 0 ;
320 switch(I2C_Flag)
321 {
322 case I2C_BUSY_FLAG :
323 {
324 if( (ACCESS_I2Cx(I2Cx)->SR2) & (I2C_SR2_BUSY) )
325 LOC_bitstatus = I2C_FLAG_SET;
326 else
327 LOC_bitstatus = I2C_FLAG_RESET;
328 break ;
329 }
330 case I2C_EV5 :
331 {
332 if( (ACCESS_I2Cx(I2Cx)->SR1) & (I2C_SR1_SB) )
333 LOC_bitstatus = I2C_FLAG_SET;
334 else
335 LOC_bitstatus = I2C_FLAG_RESET;
336 break;
337 }
338 case I2C_EV6 :
339 {
340 if( (ACCESS_I2Cx(I2Cx)->SR1) & (I2C_SR1_ADDR) )
341 LOC_bitstatus = I2C_FLAG_SET;
342 else
343 LOC_bitstatus = I2C_FLAG_RESET;
344 break;
345 }
346 case MASTER_BYTE_TRANSMITING :
347 {
348 /* Read The I2Cx Status Register */
349 LOC_u32Reg = ((((u16)(ACCESS_I2Cx(I2Cx)->SR2))<<16) | ((u16)(ACCESS_I2Cx(I2Cx)->SR1))) & ((u32)0xFFFFFFFF) ;
350 //Check whether last event contain THE I2C_EVENT
351 if((LOC_u32Reg & MASTER_BYTE_TRANSMITING) == MASTER_BYTE_TRANSMITING)
352 /* Success Last event contain THE I2C_EVENT */
353 LOC_bitstatus = I2C_FLAG_SET;
354 else
355 /* Error Last event is different from THE I2C_EVENT */
356 LOC_bitstatus = I2C_FLAG_RESET;
357 break;
358 }
359 case I2C_EV8 :
360 case I2C_EV8_1:
361 {
362 if( (ACCESS_I2Cx(I2Cx)->SR1) & (I2C_SR1_TXE) )
363 LOC_bitstatus = I2C_FLAG_SET;
364 else
365 LOC_bitstatus = I2C_FLAG_RESET;
366
367 break;
368 }
369 case I2C_EV8_2 :
370 {
371 if( (ACCESS_I2Cx(I2Cx)->SR1) & (I2C_SR1_BTF) )
372 LOC_bitstatus = I2C_FLAG_SET;
373 else
374 LOC_bitstatus = I2C_FLAG_RESET;
375
376 break;
377 }
378 case I2C_EV7 :
379 {
380 if( (ACCESS_I2Cx(I2Cx)->SR1) & (I2C_SR1_RXNE) )
381 LOC_bitstatus = I2C_FLAG_SET;
382 else
383 LOC_bitstatus = I2C_FLAG_RESET;
384
385 break;
386 }
387 }
388 return LOC_bitstatus ;
