223{
224 u32 TicksVal = 0 ;
225 u16 LOC_u16GetPrescalerVal = 0 ;
226 if(copy_u8DutyCycle >= 0 && copy_u8DutyCycle <= 100)
227 {
228 switch (ptr_userConfig->TimerNumber)
229 {
230 case TIMER0:
231 {
232 // Inverting
233 u16 arr[5] = {1,8,64,256,1024};
234 // INVERTING
235 LOC_u16GetPrescalerVal = ((u32)CPU_CLOCK_FREQ / (copy_u8FreqInHz * 510)); // Prescaler Value
236 /* Get Nearest Value */
237 LOC_u16GetPrescalerVal = GetNearestVal(LOC_u16GetPrescalerVal , arr , 5);
238 if(GET_BIT(TIM_TCCR0,TCCR0_COM00))
239 {
240 // OCR0 VAL
241 TicksVal = (510 - (((copy_u8DutyCycle * 510 ) / ((u16)100))))/2;
242 }
243 else
244 {
245 // NON-INVERTING
246 // DC =2*OCR0/510 *100 --> OCR0 = DC*51/20 ;
247 TicksVal = copy_u8DutyCycle * 2 ;
248 }
249 TIM_OCR0 = (u8)TicksVal ;
250 // Run Timer 0
251 TIM_TCCR0 &= ~(0x7) ;
252 TIM_TCCR0 |= LOC_u16GetPrescalerVal ;
253 break;
254
255 }
256 case TIMER1:
257 {
258 /* CHK ANTI-GLITCH MODE */
259 if(TIM1_ANTI_GLITCH == ANTI_GLITCH_ENABLE)
260 {
261 // GOTO ANTI GLITCH MODE BY
262 // SET WGM13 = 1 WGM12=0 WGM11=0 WGM10=1
263 SET_BIT(TIM_TCCR1B,TCCR1B_WGM13);
264 CLR_BIT(TIM_TCCR1B,TCCR1B_WGM12);
265 SET_BIT(TIM_TCCR1A,TCCR1A_WGM10);
266 CLR_BIT(TIM_TCCR1A,TCCR1A_WGM11);
267
268 /* CHK PIN IS OC1B PD4 BY CHKING VAL OF COMB1 FRM TCCR1A = 1*/
269 if(GET_BIT(TIM_TCCR1A,TCCR1A_COM1B1) == 1)
270 {
271 // Calculation and set OCR1A as a Top NOT This Freq During Up-Down Count so we need Half of it to calc TOP val
272 TicksVal = TIMER1_Prescaler * (u32)copy_u8FreqInHz ;
273 TicksVal = (u32)CPU_CLOCK_FREQ / TicksVal ;
274 TIM_OCR1A = (TicksVal/2) ;
275
276 // CHK INVERTING MODE BY GETTING VAL OF COMB0 FRM TCCR1A IF ==1 INVERTING
277 if(GET_BIT(TIM_TCCR1A,TCCR1A_COM1B0) == 1)
278 {
279 // Inverting
280 // Duty Cycle Calc
281 TicksVal = ((2*TicksVal) - (((copy_u8DutyCycle * (2*TicksVal) ) / ((u16)100))))/2;
282 }

287 // Duty Cycle Calc

305 SET_BIT(TIM_TCCR1B,TCCR1B_WGM13);
306 CLR_BIT(TIM_TCCR1B,TCCR1B_WGM12);

308 CLR_BIT(TIM_TCCR1A,TCCR1A_WGM11);
