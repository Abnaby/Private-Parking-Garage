74void
75MCAL_USART_init(UART_cnfg* ptr_UsartConfig)
76{
77 Global_uartcfg = ptr_UsartConfig;
78 /* USART STATE */
79 USART_UCSRB &= ~(0b11 << 3); // Clear Position at first
80 USART_UCSRB |= (((ptr_UsartConfig)->state) << 3) ; // Set USART State
81 /* Frame Configuration */
82 /* GET_MSB == 2 YES SET BIT TO 9TH DATA NO CLEAR BIT */
83 (((ptr_UsartConfig->databits >> 2) & 0b001) == 1) ? (USART_UCSRB |= (UCSRB_UCSZ2)) : (USART_UCSRB &= ~(UCSRB_UCSZ2));
84
85 USART_UCSRC = (UCSRC_URSEL) // SET FRAME FORMAT
86 | ((ptr_UsartConfig->databits & 0b011) << 1 ) // get bit num 2, 3 in 0bxxx then shift it to bit number 1 in UCSRC
87 | (ptr_UsartConfig->parity) // Set Parity Bits
88 | (ptr_UsartConfig->stopbits) ; // Set Number of Stop Bits
89 /* SET BAUDRATE */
90 USART_UBRRL = (u8)BAUD_PRESCALE(ptr_UsartConfig->baudrate) ;
91 USART_UBRRH = (u8)BAUD_PRESCALE(ptr_UsartConfig->baudrate) >> 8;
92}

103void
104MCAL_USART_TxSynch(u32 copy_u32Data)
105{
106 /* CHECK READY TO SEND OR NOT */
107 WAIT_TO_TRANSMIT(); /* Wait for empty transmit buffer*/
108 if(Global_uartcfg->databits == _8BIT )
109 {
110 u32 data = copy_u32Data & 0x1FF ;
111 /*
112 TXB8 is the ninth data bit in the character to be transmitted when operating with serial frames with nine data
113 bits. Must be written before writing the low bits to UDR.
114 */
115 /* Copy 9th bit to TXB8 */
116 (data = data >> 8) ? (USART_UCSRB |= UCSRB_TXB8) : (USART_UCSRB &= ~UCSRB_TXB8) ;;
117 /* Put data into buffer, sends the data */
118 USART_UDR = copy_u32Data & 0xFF0;
119 }
120 else
121 {
122 u8 *LOC_u8DataBuffer = (u8*) &copy_u32Data ;
123 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[0]);
124 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[1]);
125 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[2]);
126 MCAL_USART_TxByteSynch(LOC_u8DataBuffer[3]);
127 }

137void
138MCAL_USART_RxSynch(u32 *ptr_u32ReceivedData)
139{
140 /* WAIT UNTIL RECIEVE COMPLATE */
141 WAIT_TO_RECIEVE(); /* Wait till data is received */
142 if(Global_uartcfg->databits == _8BIT )
143 {
144 /* Get status and 9th bit, then data */
145 /* from buffer */
146 u8 resh = USART_UCSRB;
147 u8 resl = USART_UDR;
148 /* Filter the 9th bit, then return */
149 resh = (resh >> 1) & 0x01;
150 *ptr_u32ReceivedData = ((resh << 8) | resl);
151 }
152 else
153 {
154 u32 LOC_u32RecievedData = 0 ;
155 u8* LOC_u8PtrToData = (u8 *)&LOC_u32RecievedData ;
156 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[0]);
157 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[1]);
158 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[2]);
159 MCAL_USART_RxByteSynch(&LOC_u8PtrToData[3]);
160 *ptr_u32ReceivedData = LOC_u32RecievedData;

236USART_FlagsStatus MCAL_USART_GetFlagState(u8 cpy_u8Flag)
237{
238 u8 LOC_u8Result ;
239 switch(cpy_u8Flag)
240 {
241 case USART_RXC_FLAG :
242 LOC_u8Result = ((USART_UCSRA & UCSRA_RXC) >> 7)? USART_RX_COMPLATE : USART_RX_NOT_COMPLATE ;
243 break ;
244 case USART_TXC_FLAG :
245 LOC_u8Result = ((USART_UCSRA & UCSRA_TXC) >> 6)? USART_TX_COMPLATE : USART_TX_NOT_COMPLATE ;
246 break ;
247 case USART_DRE_FLAG:
248 LOC_u8Result = ((USART_UCSRA & UCSRA_UDRE) >> 5)? USART_Data_Register_Empty : USART_Data_Register_NOT_Empty ;
249 break ;
250 case USART_DATA_OVER_RUN_FLAG:
251 LOC_u8Result = ((USART_UCSRA & UCSRA_DOR) >> 3)? USART_DOV_ERROR : USART_NO_DOV_ERROR ;
252 break ;
253 case USART_PARITY_ERROR_FLAG:
254 LOC_u8Result = ((USART_UCSRA & UCSRA_PE) >> 2)? USART_PARITY_ERROR : USART_NO_PARITY_ERROR ;
255 break ;
256 case USART_FRAME_ERROR_FLAG:
257 LOC_u8Result = ((USART_UCSRA & UCSRA_FE) >> 4)? USART_FR : USART_NO_FR ;
258 break ;
259 }
260 return LOC_u8Result ;
